price = pd.read_csv('adj_close_price_weekly.csv').set_index('date')
price.index = pd.to_datetime(price.index)
price = price.fillna(method='ffill')
price = price.loc[price.index.dayofweek == 4]
In [3]:
nearness = price / price.rolling(window=52, min_periods=52, center=False).max()
In [4]:
momentum = (price - price.shift(52)) / price.shift(52)
In [5]:
# start from 2000
capital = 1000000
capital_record = {price.index[113]: capital}
for d in xrange(113, len(price.index)-4, 4):
    longtickers = []
    shorttickers = []
    try:
        nearnessquintile = pd.qcut(nearness.iloc[d], 5, labels=range(5))
    except ValueError:
        continue
    for i in range(5):
        momentumdecile = pd.qcut(momentum.iloc[d][nearnessquintile == i], 10, labels=range(10))
        longtickers += list(momentumdecile[momentumdecile == 9].index)
        shorttickers += list(momentumdecile[momentumdecile == 0].index)
    
    portfolio_value = 0.0 # the portfolio value after 4 weeks
    for t in longtickers:
        portfolio_value += price.iloc[d+4][t] * (capital / len(longtickers) / price.iloc[d][t])
    for t in shorttickers:
        portfolio_value -= price.iloc[d+4][t] * (capital / len(shorttickers) / price.iloc[d][t])
    capital += portfolio_value
    capital_record[price.index[d+4]] = capital
In [6]:
capital_trend = pd.Series(capital_record)
